# node.js操作数据库


作者：Seiya

时间：2019年05月28日


# node.js操作MySql


# 1. 安装 mysql 包

$ npm install mysql

# 2. 连接数据库


let mysql = require('mysql');

// 创建连接
let connection = mysql.createConnection({
  host : 'localhost',
  user : 'root',
  password : '123456',
  port: '3306',
  database : 'Demo'
});

// 开始连接
connection.connect();

// 执行sql语句
let sql = 'select * from users;';
connection.query(sql, (err, result) => {
	if(err) {
		console.error(err);
		return ;
	}
	console.log(result);
})


# 3. 数据库操作( CURD )


# 查询数据

具体查询语法如下：

let sql = 'select * from users;';
let sql = 'select id,username from users;';

# 插入数据

向数据表插入数据的语法具体如下：

let sql = 'INSERT INTO users(username, password, realname) VALUES("test", "123456", "测试")';

这里也可以使用另外的写法：

let sql = 'INSERT INTO users(username, password, realname) VALUES(?, ?, ?)';
let param = ["test", "123456", "测试"];
connection.query(sql, param, (err, result) => {
	...
})

# 更新数据

let sql = 'UPDATE users SET username = ?, password = ?, realname = ? where username = "zhangsan"';
let param = ["test", "123456", "测试"];
connection.query(sql, param, (err, result) => {
	...
})

# 删除数据

let sql = 'DELETE FROM users where username = "test"';
connection.query(sql, (err, result) => {
	...
})


# node.js操作MongoDB


MongoDB是一种文档导向数据库管理系统，由C++撰写而成。


# 1. 安装

$ npm install mongodb

# 2. Mac系统安装MongoDB

  • 通过 homebrew 进行安装
brew install mongodb			# homebrew安装
brew install mongodb --devel	# 安装最新版本
  • 运行 MongoDB
sudo mongod

# 如果没有创建全局路径 PATH，需要进入以下目录(mongodb安装目录)
cd /usr/local/mongodb/Cellar/bin
sudo ./mongod

# 使用一下命令进入shell模式
sudo ./mongo

# 3. node.js连接MongoDB

首先介绍 MongoDB 的标准 URI 连接语法：

mongodb://username:password@host:port/databaseName

使用 node.js 连接 MongoDB：

let client = require('mongodb').MongoClient;

// MongoDB地址（未指定用户则默认启动test数据库）
let url = "mongodb://localhost:27017";

// 连接数据库
client.connect(url, { useNewUrlParser: true }, function(err, db) {
	...
});

如果数据库不存在，MongoDB 将创建数据库并建立连接。如下：

let url = "mongodb://localhost:27017/newbase";
client.connect(url, { useNewUrlParser: true }, function(err, db) {
	if (err) throw err;
	db.close();
});

# 4. 创建集合


可以使用 createCollection() 方法来创建集合：

let url = "mongodb://localhost:27017/newbase";
client.connect(url, { useNewUrlParser: true }, function(err, db) {
	if (err) throw err;
    console.log('数据库已创建');
    var dbase = db.db("runoob");
    dbase.createCollection('test', function (err, res) {
        if (err) throw err;
        db.close();
});

# 5. 数据库操作

与 MySQL 不同的是 MongoDB 会自动创建数据库和集合，所以使用前我们不需要手动去创建。


# 插入数据

使用 insertOne() 插入一条数据条数据：

let client = require('mongodb').MongoClient;
let url = "mongodb://localhost:27017/";

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    var myobj = { name: "test", url: "test.com" };
    dbo.collection("site").insertOne(myobj, function(err, res) {
        if (err) throw err;
        db.close();
    });
});

使用 insertMany() 插入一条数据条数据：

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    var myobj =  [
        { name: 'wx', url: 'https://c.wx.com', type: 'cn'},
        { name: 'Google', url: 'https://www.google.com', type: 'en'},
        { name: 'Facebook', url: 'https://www.google.com', type: 'en'}
       ];
    dbo.collection("site").insertMany(myobj, function(err, res) {
        ...
    });
});

# 查询数据

可以使用 find() 来查找数据, find() 可以返回匹配条件的所有数据。 如果未指定条件，find() 返回集合中的所有数据。

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
     var whereStr = {"name":'test'};  // 查询条件
    dbo.collection("site").find(whereStr).toArray(function(err, result) {
        if (err) throw err;
        console.log(result);
        db.close();
    });
});

# 更新数据

更新一条数据:

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    var whereStr = {"name":'菜鸟教程'};  // 查询条件
    var updateStr = {$set: { "url" : "https://www.runoob.com" }};
    dbo.collection("site").updateOne(whereStr, updateStr, function(err, res) {
        if (err) throw err;
        db.close();
    });
});

如果要更新所有符合条的文档数据可以使用 updateMany()：

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    var whereStr = {"name":'菜鸟教程'};  // 查询条件
    var updateStr = {$set: { "url" : "https://www.runoob.com" }};
    dbo.collection("site").updateMany(whereStr, updateStr, function(err, res) {
        if (err) throw err;
        db.close();
    });
});

# 删除数据

删除一条数据：

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    var whereStr = {"name":'菜鸟教程'};  // 查询条件
    dbo.collection("site").deleteOne(whereStr, function(err, obj) {
        if (err) throw err;
        db.close();
    });
});

如果要删除多条语句可以使用 deleteMany() 方法：

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    var whereStr = {"name":'菜鸟教程' };  // 查询条件
    dbo.collection("site").deleteMany(whereStr, function(err, obj) {
        if (err) throw err;
        db.close();
    });
});

# 排序

排序 使用 sort() 方法，该方法接受一个参数，规定是升序(1)还是降序(-1)：

{ type: 1 }  // 按 type 字段升序
{ type: -1 } // 按 type 字段降序

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    var mysort = { type: 1 };
    dbo.collection("site").find().sort(mysort).toArray(function(err, result) {
        if (err) throw err;
        console.log(result);
        db.close();
    });
});

# 查询分页

如果要设置指定的返回条数可以使用 limit() 方法，该方法只接受一个参数，指定了返回的条数：

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    dbo.collection("site").find().limit(2).toArray(function(err, result) {
        if (err) throw err;
        console.log(result);
        db.close();
  });
});

如果要指定跳过的条数，可以使用 skip() 方法。

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    dbo.collection("site").find().skip(2).limit(2).toArray(function(err, result) {
        if (err) throw err;
        console.log(result);
        db.close();
  });
});

# 删除集合

使用 drop() 方法来删除集合：

client.connect(url, { useNewUrlParser: true }, function(err, db) {
    if (err) throw err;
    var dbo = db.db("runoob");
    dbo.collection("test").drop(function(err, delOK) {  // 执行成功 delOK 返回 true，否则返回 false
        if (err) throw err;
        if (delOK) console.log("集合已删除");
        db.close();
    });
});


# 6. MongoDB连接操作

mongoDB 不是一个关系型数据库，但我们可以使用 $lookup 来实现左连接。例如我们有两个集合数据分别为：

// orders
[
  { _id: 1, product_id: 154, status: 1 }
]

// product
[
  { _id: 154, name: '笔记本电脑' },
  { _id: 155, name: '耳机' },
  { _id: 156, name: '台式电脑' }
]

client.connect(url, { useNewUrlParser: true }, function(err, db) {
  if (err) throw err;
  var dbo = db.db("runoob");
  dbo.collection('orders').aggregate([
    { $lookup:
       {
         from: 'products',            // 右集合
         localField: 'product_id',    // 左集合 join 字段
         foreignField: '_id',         // 右集合 join 字段
         as: 'orderdetails'           // 新生成字段（类型array）
       }
     }
    ]).toArray(function(err, res) {
    if (err) throw err;
    console.log(JSON.stringify(res));
    db.close();
  });
});
